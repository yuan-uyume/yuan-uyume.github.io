# python 中对象的定义方法 （）中为继承于哪个父类，python与js、java一样所有的类都最终继承于object
class MyClass(object):
    cname = "aaaa"
    msg = "qqqq"
#     def __init__(self):
#         self.cname = cname
#         self.msg = msg
    #初始化函数__init__在类实例化时调用，且不能被重载
    def __init__(self,cname,msg):
        self.cname = cname
        self.msg = msg
        self.__v = "1111"
    # python中没有绝对的权限控制，默认使用_前缀来提醒使用者此为私有变量不要调用，类似的还有用大写变量表示常数。
    # __为前缀则表示为私有变量，但是只是解释器将 __*变成了_classname__*，来控制访问，不同版本的解释器可能变换规则不同
    __version = "0.112"
    # 在python的类中，定义的函数一个参数将接受实例本身
    def getVersion(self):
        return self.__version
mc = MyClass('init','in')
ver = mc.getVersion()
print(mc.msg)
# 绕过限制直接访问私有变量__version
print(mc._MyClass__version)
print(ver)
# python中的类接近于js中的对象
mc.addp = "add"
print(mc.addp)
# 此时是直接向类实例对象中添加变量
mc.__vadd = "1111"
print(mc.__vadd)
# __init__中设置的变量也会被转化名字
print(mc.__v)
# 直接输出报的错误是 没有该属性，所以私有属性只不过是在实例化时将变量名转换了
print(mc.__version)

# python 中的继承和多态

python 相较于 java 等静态语言，他并不要求严格的继承体系，一个对象只要 “看起来像鸭子，走起路来像鸭子”，那它就可以被看做是鸭子。<br>
对于静态语言（例如 Java）来说，如果需要传入 Animal 类型，则传入的对象必须是 Animal 类型或者它的子类，否则，将无法调用 run () 方法。<br>
Python 的 “file-like object“就是一种鸭子类型。对真正的文件对象，它有一个 read () 方法，返回其内容。但是，许多对象，只要有 read () 方法，都被视为 “file-like object“。许多函数接收的参数就是 “file-like object“，你不一定要传入真正的文件对象，完全可以传入任何实现了 read () 方法的对象。

# 判断一个普通对象的类型
l = [11111]
print(type(l))
# 判断一个继承对象的类型
print(isinstance(l,list))
print(isinstance(l,object))
# 输出对象的所有属性
print(dir(l))
# len(l) 等价于 l.__len__()
print(l.__len__())
# 对象属性获取、设置或判断
print(hasattr(l,"__len__"))
print(getattr(l,'__len__'))

# 实例属性和类属性
class Oneclass(object):
    name = 'classAttr'
    def __init__(self):
        self.name = 'entityAttr'
oc = Oneclass()
# 实例属性优先于类属性显示，当实例属性不存在或被删除时将显示实例属性
print(oc.name)
# 删除实例的name属性
del oc.name
print(oc.name)
