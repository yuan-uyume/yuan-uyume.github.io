# map()函数,两个参数第一个参数接受一个函数，第二个参数接受一个Iterable对象
ay = [1,2,3,4,5,6]
ty = (1,2,3,4,5,6)
def f(x):
    return x*x
# map()函数返回的是一个Iterator,迭代器
ma = map(f,ay)
mt = map(f,ty)
print(list(ma))
print(mt)
# Iterator对象只能被遍历一次，原因是他是数据流
for a in ma:
    print(a)
for t in mt:
    print(t)
print(tuple(mt))

这是因为 Python 的 Iterator 对象表示的是一个数据流，Iterator 对象可以被 next () 函数调用并不断返回下一个数据，直到没有数据时抛出 StopIteration 错误。可以把这个数据流看做是一个有序序列，但我们却不能提前知道序列的长度，只能不断通过 next () 函数实现按需计算下一个数据，所以 Iterator 的计算是惰性的，只有在需要返回下一个数据时它才会计算。

# reduce() 套娃递归函数,接受一个函数和一个序列，返回递归结果. python3中从全局函数中移除了，需要functools中引入
from functools import reduce
a = [1,2,3,4,5,6]
t = (1,2,3,4,5,6)
def f(x,y):
    return x+y
print(reduce(f,a))
print(reduce(f,t))

# filter() 用于过滤数组，第一个参数是过滤函数，true保存false丢弃。第二个参数接受数组
a = [1,2,3,4,5,6,7,8]
# 可以使用lambda来使用匿名函数，同时0可表False，1可表True
print(list(filter(lambda x: True if x%2 == 0 else False ,a)))
print(list(filter(lambda x: x%2,a)))

[1, 3, 5, 7]

# sorted函数,key函数作用于每个元素后在进行排序

# 函数作为返回值
def lazy_sum(*args):
    def sum():
        ax = 0
        for n in args:
            ax = ax + n
        return ax
    return sum
f = lazy_sum(1,2,3,4)
ff = lazy_sum(1,2,3,4)
print(f)
print(f())
print(f == ff)

# 闭包问题
def count():
    fs = []
    for i in range(1, 4):
        def f():
             return i*i
        fs.append(f)
    return fs

f1, f2, f3 = count()
print(f1())
print(f2())
print(f3())

# 解决方案
def count():
    def g(j):
        def f():
            return j*j
        return f
    fs = []
    for i in range(1, 4):
        fs.append(g(i))
    return fs
f1, f2, f3 = count()
print(f1())
print(f2())
print(f3())

# 使用lambda 简化
def count():
    f = lambda j: lambda: j*j
    fs = []
    for i in range(1, 4):
        fs.append(f(i))
    return fs

# 装饰器（decorator）,本质为一个返回函数的函数。接受一个函数，然后在源函数前后添加语句返回修饰后的函数。
# 即返回一个新的函数，新的函数中调用了传入的函数，不改变原来的函数
def log(func):
    def wrapper(*args, **kw):
        print('call %s():' % func.__name__)
        return func(*args, **kw)
    return wrapper
@log
def now():
    print('2015-3-25')
now()

#同
def now1():
    print('2077-7-7')
now1 = log(now1)
now1()

# 前后同时修饰
def log2(func):
    def wrapper():
        print('call %s():' % func.__name__)
        func()
        print('call %s() end:' % func.__name__)
    return wrapper
@log2
def now2():
    print('2015-3-25')
now2()

call now():
2015-3-25
call now1():
2077-7-7
call now2():
2015-3-25
call now2() end:

# Python的functools模块提供了很多有用的功能，其中一个就是偏函数（Partial function）
# 给源函数固定参数默认值或者添加一个参数，返回修改的后的参数
from functools import partial
# 给max函数固定添加一个参数 10
max2 = partial(max,10)
print(max2(1,2,3,4))
# 给int函数固定base参数为 2
int2 = partial(int, base = 2)
print(int2("101"))
print(int2("101", base = 10))
